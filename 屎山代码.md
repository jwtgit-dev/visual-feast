## 1.插入排序
```c
 #include <stdio.h>
void insertsort(int a[],int n);
int main(){
	int a[5];
	int i;
	for(i=0;i<5;i++){
		scanf("%d",a+i);
	}
	insertsort(a,5);
	for(i=0;i<5;i++){
		printf("%d ",a[i]);
	}
	
	return 0;
}
void insertsort(int a[],int n){
	int i,j;
	int key;
	for(i=1;i<n;i++){
		key=a[i];
		for(j=i-1;j>=0;j--){
			if(key<a[j]){
				a[j+1]=a[j];	
			}
		}
		a[j+1]=key;
	}
}
```
## 2.倒序输出
```c
#include <stdio.h>
void upsidedown(int a[],int len);
int main (){
	int a[5]={0};
	int i;
	for(i=0;i<5;i++){
		scanf("%d",a+i);
	}
	upsidedown(a,5);
	for(i=0;i<5;i++){
		printf("%d ",a[i]);
	}
	
	return 0;
}
void upsidedown(int a[],int len){
	int*p=a;
	int*q=a+len-1;
	int temp;



	for(;p<q;p++,q--){
		temp=*p;
		*p=*q;
		*q=temp;
	}
}
```

## 4.两个查找函数

```c
int sequentialSearch(int a[], int n, int target) {
   
    for (int i = 0; i < n; i++) {
        
        if (a[i] == target) {
            return i;
        }
    }

    return -1;
}







int binarySearch(int a[], int n, int target) {
    int left = 0;          
    int right = n - 1;     
    int mid;              


    while (left <= right) {
     
        mid = left + (right - left) / 2;
        
        if (a[mid] == target) {
        
            return mid;
        } else if (a[mid] < target) {
         
            left = mid + 1;
        } else {
       
            right = mid - 1;
        }
    }

    return -1;
}
```
## 5.冒泡排序
```c
#include <stdio.h>
void bublesort(int a[],int n);
int main(){
	int a[5],i;
	for(i=0;i<5;i++){
		scanf("%d",a+i);
	}
	bublesort(a,5);
	for(i=0;i<5;i++){
		printf("%d ",a[i]);
	}
	
	
	
	
	return 0;
}
void bublesort(int a[],int n){
	int i,j,temp;
	for(i=0;i<n;i++){
		for(j=0;j<n-i-1;j++){
			if(a[j]>a[j+1]){
				temp=a[j];
				a[j]=a[j+1];
				a[j+1]=temp;
			}
		}
	}
}
```
## 6.每月的天数
```c
#include <stdio.h>

int isLeapYear(int year);
int getDaysOfMonth(int year, int month);

int main() {
    int year, month;
    scanf("%d %d", &year, &month);
    
    int days = getDaysOfMonth(year, month);
    printf("%d\n", days);
    
    return 0;
}

int isLeapYear(int year) {
    return (year % 4 == 0 && year % 100 != 0) || (year % 400 == 0);
}

int getDaysOfMonth(int year, int month) {
    switch (month) {
        case 1:
        case 3:
        case 5:
        case 7:
        case 8:
        case 10:
        case 12:
            return 31;
        case 4:
        case 6:
        case 9:
        case 11:
            return 30;
        case 2:
            return isLeapYear(year) ? 29 : 28;
        default:
            return -1;
    }
}
```

## 9.选择排序
```c
#include <stdio.h>
void selectsort(int a[],int n);
int main (){
	int a[5];
	int i;
	
	for(i=0;i<5;i++){
		scanf("%d",a+i);
	}
	selectsort(a,5);
	for(i=0;i<5;i++){
		printf("%d ",a[i]);
	}
	
	
	return 0;
}
void selectsort(int a[],int n){
	int i,j,minindex,temp;
	for(i=0;i<n-1;i++){
		minindex=i;
		for(j=i+1;j<n;j++){
			if(a[j]<a[minindex]){
				minindex=j;
			}
		}
		temp=a[i];
		a[i]=a[minindex];
		a[minindex]=temp;
	}
}

```
